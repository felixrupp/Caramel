  728.     /**
  729.      * Method to create a new page with one default record
  730.      * 
  731.      * @param string $path URL path of the new page
  732.      * @param string $defaultLang Language for default record of the new page
  733.      * @param array $recordContents Array with contents of default record
  734.      * 
  735.      * @throws CaramelException
  736.      * @return Result of file_put_contents
  737.      */
  738.     public function createPageAction($path, $defaultLang, $recordContents) {
  739.         
  740.         $result = false;
  741.         $idArray = array();
  742.         
  743.         $xPathId = $this->_dataBase->xpath("page/@id");
  744.                 
  745.         if(count($xPathId)>0) {
  746.             
  747.             foreach($xPathId as $idNode) {
  748.                 $idArray[] = (int)$idNode["id"];
  749.             }
  750.             
  751.             $newID = max($idArray)+1;
  752.             
  753.             
  754.             # Add page
  755.             $newPage = $this->_dataBase->addChildCData("page");
  756.             
  757.             ## Set ID
  758.             $newPage->addAttribute("id", $newID);
  759.             
  760.             ## Set path
  761.             $newPage->addAttribute("path", $path);
  762.             
  763.             
  764.             # Add record
  765.             $newRecord = $newPage->addChildCData("record");
  766.             
  767.             ## Set language
  768.             $newRecord->addAttribute("lang", $defaultLang);
  769.             
  770.             ## Add rest of children
  771.             $navigation = $newRecord->addChildCData("navigation", $recordContents["navigation"]);
  772.             
  773.             $newRecord->addChild("title", $recordContents["title"]);
  774.             
  775.             $newRecord->addChild("titletag", $recordContents["titletag"]);
  776.             
  777.             $metaDescription = $newRecord->addChildCData("meta", $recordContents["metadescription"]);
  778.             $metaDescription->addAttribute("name", "description");
  779.             
  780.             $metaKeywords = $newRecord->addChildCData("meta", $recordContents["metakeywords"]);

  782.             
  783.             $metaAuthor = $newRecord->addChildCData("meta", $recordContents["metaauthor"]);

  786.             $newRecord->addChildCData("socialbar", "false");

  788.             $newRecord->addChildCData("content", $recordContents["content"]);
  789.             
  790.             $result = file_put_contents(BASEDIR.'/database/data.xml', $this->_dataBase->asXML());

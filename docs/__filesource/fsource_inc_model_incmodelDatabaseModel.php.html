  260.             }

  268.             foreach($subPages as $subPage) {
  269.                 
  270.                 ## Get subrecords

  273.                 if(count($xPathResultRecord) > 0) {

  275.                     $subRecord = $xPathResultRecord[0];

  277.                     $orderedNavi[(int)$page->attributes()->id]["subpages"][(int)$subPage->attributes()->id] = array(
  278.                         "path" => (string)$subPage->attributes()->path,
  279.                         "pos" => (int)$subRecord->navigation->attributes()->pos,

  285.                 }

  287.                 
  288.             }

  290.             
  291.         }
